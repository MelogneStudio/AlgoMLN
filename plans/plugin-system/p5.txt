pub enum EventFilter {
  All,
  CandleProcessed,
  TradeExecuted,
  RuleFired,
  StrategyStatusChanged,
  SystemShutdown,
}
impl EventFilter {
  fn matches(&self, event: &EventKind) -> bool
    All matches everything. Others match only their discriminant.
}

pub struct EventBus {
  subscribers: std::sync::Arc
    parking_lot::RwLock
      Vec<(
        crate::plugin::SubscriptionHandle,
        EventFilter,
        std::sync::Arc<dyn Fn(EventKind) + Send + Sync>,
      )>
    >
  >,
}
impl EventBus {
  pub fn new() -> std::sync::Arc<Self>
  pub fn subscribe(
    &self,
    filter: EventFilter,
    callback: std::sync::Arc<dyn Fn(EventKind) + Send + Sync>,
  ) -> crate::plugin::SubscriptionHandle
    Generates handle, pushes to subscribers, returns handle.
  pub fn unsubscribe(&self, handle: &crate::plugin::SubscriptionHandle)
    Removes the entry with matching handle.
  pub fn publish(&self, event: EventKind)
    Acquires read lock. For each matching subscriber, clone the event and spawn a tokio task
    that calls the callback. Release the lock before spawning.
}

  >>,
}

Implement SchedulerApi:
  schedule:
    Parse cron_expr with cron::Schedule::from_str.
    Map parse error to PluginError::ApiError.
    Create a CancellationToken.
    Spawn a tokio task:
      Loop: get next scheduled time from schedule.upcoming(chrono::Utc).next().
        Sleep until that time using tokio::time::sleep_until.
        If cancelled (select! on token.cancelled()), break.
        Call callback().
    Store handle → token in map. Return handle.
  cancel:
    Look up token, call token.cancel(), remove from map.
    Return PluginError::NotFound if handle not present.
  cancel_all_for:
    No per-plugin tracking here — callers are responsible for passing the correct handles.
    (PluginRegistry will store handles per plugin_id separately.)

Add chrono to Cargo.toml if not present.
