Create src/commands/plugins.rs:

use tauri::State;
use crate::plugin::{PluginId, PluginListEntry};
use crate::commands::AppState;

#[tauri::command]
pub async fn list_plugins(state: State<'_, AppState>) -> Result<Vec<PluginListEntry>, String> {
  Ok(state.plugin_registry.list())
}

#[tauri::command]
pub async fn enable_plugin(id: String, state: State<'_, AppState>) -> Result<(), String> {
  state.plugin_registry.enable(&PluginId(id)).await.map_err(|e| e.to_string())
}

#[tauri::command]
pub async fn disable_plugin(id: String, state: State<'_, AppState>) -> Result<(), String> {
  state.plugin_registry.disable(&PluginId(id)).await.map_err(|e| e.to_string())
}

#[tauri::command]
pub async fn reload_plugins(state: State<'_, AppState>) -> Result<Vec<String>, String> {
  let results = state.plugin_registry.scan_and_load().await;
  Ok(results.into_iter().filter_map(|(id, r)| r.err().map(|e| format!("{}: {}", id, e))).collect())
}
